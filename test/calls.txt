# No Parameters

func()

==>

Recipe(Call(ArgList))

# One Variable

f( x )

==>

Recipe(Call(ArgList(Variable)))

# Two Variables

f(x,y)

==>

Recipe(Call(ArgList(Variable, Variable)))

# Three Variables

f(x, y, z)

==>

Recipe(Call(ArgList(Variable, Variable, Variable)))

# Nested Call

f(g())

==>

Recipe(Call(ArgList(Call(ArgList))))

# Nested Nested Call

f( g( h() ) )

==>

Recipe(Call(ArgList(Call(ArgList(Call(ArgList))))))

# Variable and Call

f ( x , g ( ) )

==>

Recipe(Call(ArgList(Variable, Call(ArgList))))

# Call and Variable

f (
  g ( ) ,
  y
)

==>

Recipe(Call(ArgList(Call(ArgList), Variable)))

# Nested Nested Call with Variable Parameters

f(
  g(
    h(x,y),
    z
  )
)

==>

Recipe(
  Call(ArgList(
    Call(ArgList(
      Call(ArgList(Variable, Variable)),
      Variable
    ))
  ))
)
